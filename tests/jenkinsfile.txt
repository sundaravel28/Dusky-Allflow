pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/sundaravel28/Dusky-Allflow.git'
            }
        }

        stage('Install Dependencies') {
            steps {
                bat 'npm install'
            }
        }

        stage('Run Tests') {
            steps {
                bat 'npx playwright test'   // or your test command
            }
        }
    }

    post {
        always {
            echo 'Build finished.'
        }
    }
}
